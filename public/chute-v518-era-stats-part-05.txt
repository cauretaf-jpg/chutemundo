  const renderSignature = JSON.stringify([
    selectedEra,
    selectedTournament,
    selectedTeam,
    tournaments.map((tournament) => [
      tournament.id, tournament.eraId, tournament.status, tournament.champion, tournament.runnerUp, tournament.third, tournament.coverage,
      (tournament.matches || []).map((match) => [match.id, match.homeGoals, match.awayGoals, match.homePens, match.awayPens, (match.goals || []).length, (match.cards || []).length, match.participationTracked, match.lineups, match.penaltyShootout])
    ]),
    (source.teams || []).map((team) => [team.id, team.name, team.players]),
    source.participants
  ]);
  if (host.dataset.signature === renderSignature) { activateTab(selectedTab); return; }
  host.dataset.signature = renderSignature;
  const tournamentOptions = (source.tournaments || []).filter((tournament) => selectedEra === 'all' || tournament.eraId === selectedEra).map((tournament) => `<option value="${esc(tournament.id)}" ${selectedTournament === tournament.id ? 'selected' : ''}>${esc(tournament.name)}</option>`).join('');
  const teamOptions = (source.teams || []).map((team) => `<option value="${esc(team.id)}" ${selectedTeam === team.id ? 'selected' : ''}>${esc(team.name)}</option>`).join('');
  const tabs = [['summary','Resumen'],['teams','Equipos'],['players','Jugadores'],['keepers','Porteros'],['tournaments','Torneos'],['history','Historia']];
  host.innerHTML = `<section class="cm-v518-toolbar">
      <div class="cm-v518-filter-grid">
        <label>Periodo<select data-cm-v518-filter="era"><option value="all" ${selectedEra === 'all' ? 'selected' : ''}>Toda la historia</option><option value="${ERA_LEAGUES}" ${selectedEra === ERA_LEAGUES ? 'selected' : ''}>Era de ligas</option><option value="${ERA_DIVISIONS}" ${selectedEra === ERA_DIVISIONS ? 'selected' : ''}>Era de divisiones</option></select></label>
        <label>Torneo<select data-cm-v518-filter="tournament"><option value="all">Todos los torneos</option>${tournamentOptions}</select></label>
        <label>Equipo<select data-cm-v518-filter="team"><option value="all">Todos los equipos</option>${teamOptions}</select></label>
      </div>
      <nav class="cm-v518-tabs" aria-label="Secciones estadísticas">${tabs.map(([id, label]) => `<button class="${selectedTab === id ? 'active' : ''}" data-cm-v518-tab="${id}">${label}</button>`).join('')}</nav>
    </section>
    <div class="cm-v518-content">
      ${summaryPanel(tournaments, source)}
      ${teamsPanel(tournaments, source)}
      ${playersPanel(tournaments, source)}
      ${keepersPanel(tournaments, source)}
      ${tournamentsPanel(tournaments)}
      ${historyPanel(tournaments, source)}
    </div>`;
  activateTab(selectedTab);
}

function activateTab(tab) {
  selectedTab = ['summary','teams','players','keepers','tournaments','history'].includes(tab) ? tab : 'summary';
  localStorage.setItem('cm_v518_stats_tab', selectedTab);
  document.querySelectorAll('#cmV518Stats [data-cm-v518-tab]').forEach((button) => button.classList.toggle('active', button.dataset.cmV518Tab === selectedTab));
  document.querySelectorAll('#cmV518Stats [data-cm-v518-panel]').forEach((panel) => panel.classList.toggle('active', panel.dataset.cmV518Panel === selectedTab));
}

function decorateTournamentForm() {
  const form = document.getElementById('tournamentForm');
  if (!form || form.querySelector('[data-cm-v518-era-note]')) return;
  const notice = document.createElement('div');
  notice.dataset.cmV518EraNote = '';
  notice.className = 'notice info cm-v518-era-note';
  notice.innerHTML = '<b>Era estadística: Era de divisiones</b><span>Los nuevos torneos exigirán goleadores, asistencias o “sin asistencia”, alineaciones, minutos, cambios, tarjetas, sede y tandas completas.</span>';
  form.querySelector('#formatHelp')?.insertAdjacentElement('afterend', notice);
}

function openDivisionsBlock(tournament, issues) {
  core.openModal(`<section class="cm-v518-quality"><p class="eyebrow">ERA DE DIVISIONES</p><h2>No se puede cerrar todavía</h2><p>La nueva era exige información completa para que sus estadísticas sean comparables.</p><ul>${issues.slice(0, 30).map((issue) => `<li>${esc(issue)}</li>`).join('')}</ul><div class="modal-actions"><button class="secondary" data-close-modal>Volver</button></div></section>`);
}

function installStyles() {
  if (document.getElementById('cmV518Styles')) return;
  const link = document.createElement('link');
  link.id = 'cmV518Styles';
  link.rel = 'stylesheet';
  link.href = `/chute-v518-era-stats.css?v=${VERSION}`;
  document.head.appendChild(link);
}

function refresh() {
  refreshQueued = false;
  void ensureMigration();
  decorateTournamentForm();
  renderShell();
  document.title = 'Chute Mundo v5.18 · Eras y estadísticas';
  document.querySelector('.hero .eyebrow')?.replaceChildren('CHUTE MUNDO v5.18');
}

function scheduleRefresh() {
  if (refreshQueued) return;
  refreshQueued = true;
  requestAnimationFrame(refresh);
}

window.addEventListener('click', (event) => {
  const confirm = event.target.closest?.('[data-cm-v517-confirm-finish]');
  if (!confirm) return;
  const tournament = core.tournamentById(confirm.dataset.cmV517ConfirmFinish);
  const issues = tournament ? divisionsCriticalIssues(tournament) : [];
  if (!issues.length) return;
  event.preventDefault();
  event.stopImmediatePropagation();
  openDivisionsBlock(tournament, issues);
}, true);

document.addEventListener('click', (event) => {
  const tab = event.target.closest?.('[data-cm-v518-tab]');
  if (tab) {
    event.preventDefault();
    activateTab(tab.dataset.cmV518Tab);
  }
  const team = event.target.closest?.('[data-cm-v518-open-team]');
  if (team) {
    core.navigate('equipos');
    window.ChuteV59?.openTeamProfile?.(team.dataset.cmV518OpenTeam);
  }
  const player = event.target.closest?.('[data-cm-v518-open-player]');
  if (player) {
    core.navigate('equipos');
    window.ChuteV59?.openPlayerProfile?.(player.dataset.cmV518OpenPlayer);
  }
}, true);

document.addEventListener('change', (event) => {
  const filter = event.target.closest?.('[data-cm-v518-filter]');
  if (!filter) return;
  if (filter.dataset.cmV518Filter === 'era') {
    selectedEra = filter.value;
    selectedTournament = 'all';
    localStorage.setItem('cm_v518_stats_era', selectedEra);
  }
  if (filter.dataset.cmV518Filter === 'tournament') {
    selectedTournament = filter.value;
    localStorage.setItem('cm_v518_stats_tournament', selectedTournament);
  }
  if (filter.dataset.cmV518Filter === 'team') {
    selectedTeam = filter.value;
    localStorage.setItem('cm_v518_stats_team', selectedTeam);
  }
  renderShell();
}, true);

document.addEventListener('submit', (event) => {
  if (event.target.id !== 'tournamentForm') return;
  setTimeout(() => { void ensureMigration(); scheduleRefresh(); }, 350);
}, true);

new MutationObserver(scheduleRefresh).observe(document.body, { childList: true, subtree: true });
installStyles();
if ('serviceWorker' in navigator) navigator.serviceWorker.register(`/sw.js?v=${VERSION}`).catch((error) => console.warn('No se pudo actualizar la PWA v5.18.', error));
scheduleRefresh();

window.ChuteV518EraStats = {
  version: VERSION,
  cutoffName: CUTOFF_NAME,
  plannedEra,
  inferCoverage,
  divisionsCriticalIssues,
  migrateState,
  teamRows,
  playerRows,
  renderShell
};
